// “естирование подсчета ссылок 
#include <cassert>
#include "DBConnection.h"

class DBClient {
  DBConnection* db;
public:
  DBClient(DBConnection* dbCon) {
    db = dbCon;
    db->attach();
  }
  ~DBClient() {
    db->detach();
  }
  // ƒругие запросы к базе данных с использованием db...
};

int main() {
  DBConnection* db = DBConnection::create("MyDatabase");
  assert(db->refCount() == 1);
  DBClient c1(db);
  assert(db->refCount() == 2);
  DBClient c2(db);
  assert(db->refCount() == 3);
  // »спользуем базу данных, а затем освобождаем ресурс,
  // захваченный при вызове create.
  db->detach();
  assert(db->refCount() == 2);
} ///:~
